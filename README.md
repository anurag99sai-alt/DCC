    "def get_daily_documents(target_date_str: str, company_ids: List[str]):\n",
    "    start_timestamp = f\"{target_date_str}T00:00:00.000Z\"\n",
    "    end_timestamp = f\"{target_date_str}T23:59:59.999Z\"\n",
    "    data_endpoint = \"/bigdata/v1/search\"\n",
    "\n",
    "    def make_request_with_retry(ids, batch_num, total_batches, max_retries=MAX_RETRIES):\n",
    "        \"\"\"Make request with exponential backoff retry logic for rate limiting\"\"\"\n",
    "        ## there are many other api points and parameters can be loaded into this payload. please be creative\n",
    "        payload = {\n",
    "            \"query\": {\n",
    "                \"filters\": {\n",
    "                    \"timestamp\": {\"start\": start_timestamp, \"end\": end_timestamp},\n",
    "                    \"entity\": {\"any_of\": ids}\n",
    "                }\n",
    "            }\n",
    "        }\n",
    "        \n",
    "        for attempt in range(max_retries):\n",
    "            response = None\n",
    "            try:\n",
    "                response = session.post(data_endpoint, json=payload)\n",
    "                response.raise_for_status()\n",
    "                return response.json().get('results', [])\n",
    "            except requests.exceptions.HTTPError as http_err:\n",
    "                # Use the response from the exception if available, or the local variable\n",
    "                err_response = getattr(http_err, 'response', response)\n",
    "                \n",
    "                if err_response is not None and err_response.status_code == 429:  # Too Many Requests\n",
    "                    # Exponential backoff: 2^attempt seconds (1, 2, 4, 8, 16 seconds)\n",
    "                    wait_time = min(2 ** attempt, MAX_BACKOFF_TIME)  # Cap at max backoff time\n",
    "                    print(f\"  Rate limit hit for batch {batch_num}/{total_batches}. Waiting {wait_time}s before retry {attempt+1}/{max_retries}...\")\n",
    "                    time.sleep(wait_time)\n",
    "                else:\n",
    "                    print(f\"  HTTP error for batch {batch_num}/{total_batches}: {http_err}\")\n",
    "                    break\n",
    "            except Exception as req_err:\n",
    "                print(f\"  Request error for batch {batch_num}/{total_batches}: {req_err}\")\n",
    "                break\n",
    "        \n",
    "        print(f\"  Failed to fetch batch {batch_num}/{total_batches} after {max_retries} attempts\")\n",
    "        return []\n",
    "\n",
    "    # Batch IDs into chunks\n",
    "    all_docs = []\n",
    "    total_batches = (len(company_ids) + BATCH_SIZE - 1) // BATCH_SIZE\n",
    "    \n",
    "    print(f\"  Processing {len(company_ids)} companies in {total_batches} batches...\")\n",
    "    \n",
    "    for i in range(0, len(company_ids), BATCH_SIZE):\n",
    "        batch_num = (i // BATCH_SIZE) + 1\n",
    "        chunk = company_ids[i:i + BATCH_SIZE]\n",
    "        \n",
    "        # Add a small delay between requests to avoid hitting rate limits\n",
    "        if batch_num > 1:\n",
    "            time.sleep(REQUEST_DELAY)  # Delay between requests\n",
    "        \n",
    "        print(f\"  Fetching batch {batch_num}/{total_batches} ({len(chunk)} companies)...\")\n",
    "        batch_docs = make_request_with_retry(chunk, batch_num, total_batches)\n",
    "        all_docs.extend(batch_docs)\n",
    "        \n",
    "    print(f\"  Completed: Retrieved {len(all_docs)} documents for {target_date_str}\")\n",
    "    return all_docs"

    "def calculate_scores_for_day(documents, valid_ids_set):\n",
    "    \"\"\"Process raw documents and return a dict of {entity_id: [scores]}\"\"\"\n",
    "    day_scores = defaultdict(list)\n",
    "    \n",
    "    for article in documents:\n",
    "        chunk_scores = []\n",
    "        article_entities = set()\n",
    "\n",
    "        # 1. Calculate average score for the article based on chunks\n",
    "        for chunk in article.get('chunks', []):\n",
    "            relevance = chunk.get('relevance', 0)\n",
    "            sentiment = chunk.get('sentiment', 0)\n",
    "            if relevance is not None and sentiment is not None:\n",
    "                score = relevance * sentiment\n",
    "                chunk_scores.append(score)\n",
    "            \n",
    "            for detection in chunk.get('detections', []):\n",
    "                if detection.get('type') == 'entity':\n",
    "                    entity_id = detection.get('id')\n",
    "                    if entity_id and entity_id in valid_ids_set:\n",
    "                        article_entities.add(entity_id)\n",
    "        \n",
    "        # 2. Assign article score to all found entities\n",
    "        if chunk_scores and article_entities:\n",
    "            article_score = round((sum(chunk_scores) / len(chunk_scores)) * 100, 2)\n",
    "            for entity_id in article_entities:\n",
    "                day_scores[entity_id].append(article_score)\n",
    "                \n",
    "    return day_scores"

    "# Execution Logic - Fetch and Process In-Memory\n",
    "\n",
    "def run_workflow(start_date_input: Union[str, date, datetime], end_date_input: Union[str, date, datetime], universe_ids: List[str]):\n",
    "    \"\"\"\n",
    "    Executes the data retrieval and scoring workflow for a given date range.\n",
    "    \n",
    "    Args:\n",
    "        start_date_input (str, date, datetime): Start date (e.g., '2021-01-01' or date object).\n",
    "        end_date_input (str, date, datetime): End date (e.g., '2021-01-07' or date object).\n",
    "        universe_ids (list): List of company IDs to process.\n",
    "        \n",
    "    Returns:\n",
    "        dict: Aggregated data structure { entity_id: { date_str: [scores] } }\n",
    "    \"\"\"\n",
    "    \n",
    "    # Helper to parse date inputs\n",
    "    def parse_date(d):\n",
    "        if isinstance(d, str):\n",
    "            return datetime.strptime(d, \"%Y-%m-%d\").date()\n",
    "        elif isinstance(d, datetime):\n",
    "            return d.date()\n",
    "        elif isinstance(d, date):\n",
    "            return d\n",
    "        else:\n",
    "            raise ValueError(f\"Invalid date format: {d} (type: {type(d)})\")\n",
    "\n",
    "    start_date = parse_date(start_date_input)\n",
    "    end_date = parse_date(end_date_input)\n",
    "    \n",
    "    start_date_str = start_date.strftime(\"%Y-%m-%d\")\n",
    "    end_date_str = end_date.strftime(\"%Y-%m-%d\")\n",
    "\n",
    "    print(f\"--- Starting daily document retrieval for {len(universe_ids)} companies ---\")\n",
    "    print(f\"--- Period: {start_date_str} to {end_date_str} ---\")\n",
    "\n",
    "    delta = timedelta(days=1)\n",
    "    valid_ids_set = set(universe_ids)\n",
    "    \n",
    "    # Structure: { entity_id: { date_str: [scores] } }\n",
    "    all_entity_data = defaultdict(lambda: defaultdict(list))\n",
    "\n",
    "    current_date = start_date\n",
    "    while current_date <= end_date:\n",
    "        date_str = current_date.strftime(\"%Y-%m-%d\")\n",
    "        print(f\"Processing data for {date_str}...\")\n",
    "        \n",
    "        # 1. Fetch\n",
    "        documents = get_daily_documents(date_str, universe_ids)\n",
    "        \n",
    "        # 2. Score\n",
    "        daily_scores = calculate_scores_for_day(documents, valid_ids_set)\n",
    "        \n",
    "        # 3. Aggregate into main structure\n",
    "        for entity_id, scores in daily_scores.items():\n",
    "            all_entity_data[entity_id][date_str] = scores\n",
    "            \n",
    "        current_date += delta\n",
    "\n",
    "    print(\"\\n--- Data processing complete ---\")\n",
    "    return all_entity_data\n",
    "\n",
    "# Run the workflow\n",
    "# Example usage with string inputs (can also be date/datetime objects)\n",
    "all_entity_data = run_workflow(start_date_input=\"2021-01-01\", end_date_input=\"2021-01-01\", universe_ids=top3000)"

    "def create_dataframe(entity_data, start_date_input, end_date_input, universe_ids, output_path=\"sentiment_data.parquet\"):\n",
    "    \"\"\"\n",
    "    Converts the aggregated entity data into a pandas DataFrame and saves it as a parquet file.\n",
    "    \"\"\"\n",
    "    # Create a list of all dates covered\n",
    "    # pd.date_range handles various input types gracefully\n",
    "    date_range = pd.date_range(start=start_date_input, end=end_date_input, freq='D')\n",
    "    sorted_dates = [d.strftime(\"%Y-%m-%d\") for d in date_range]\n",
    "\n",
    "    # Construct DataFrame\n",
    "    # Rows: Dates\n",
    "    # Columns: Entity IDs\n",
    "    # Values: List of scores (or empty list/None)\n",
    "\n",
    "    df_data = {}\n",
    "    for entity_id in universe_ids:\n",
    "        # Get scores for each date, default to None if no scores found\n",
    "        entity_daily_scores = entity_data.get(entity_id, {})\n",
    "        df_data[entity_id] = [entity_daily_scores.get(d, None) for d in sorted_dates]\n",
    "\n",
    "    df_raw = pd.DataFrame(df_data, index=pd.to_datetime(sorted_dates))\n",
    "    df_raw.index.name = 'date'\n",
    "    \n",
    "    # Save to parquet locally\n",
    "    df_raw.to_parquet(output_path)\n",
    "    print(f\"DataFrame saved to {output_path}\")\n",
    "    \n",
    "    return df_raw\n",

    "#### If you want to create a \"MATRIX\" TYPE Data\n",
    "def calc_average(scores):\n",
    "    if isinstance(scores, list) and scores:\n",
    "        return sum(scores) / len(scores)\n",
    "    return 0.0 # Default value for missing data\n",
    "\n",
    "def create_data_field(data_field_id, data_frame, region, delay, type_name, source):\n",
    "    data_frame_copy = data_frame.copy()\n",
    "    buffer = io.BytesIO()\n",
    "    table = pyarrow.Table.from_pandas(data_frame_copy, preserve_index=True)\n",
    "    parquet.write_table(table, buffer, compression='GZIP')\n",
    "    buffer.seek(0)\n",
    "\n",
    "    upload_url = \"/users/self/data-fields\"\n",
    "    \n",
    "    data = {\n",
    "        'field': json.dumps({\n",
    "            'id': data_field_id, \n",
    "            'type': type_name, \n",
    "            'source': source, \n",
    "            'data': [{'region': region, 'delay': delay}]\n",
    "        })\n",
    "    }\n",
    "    \n",
    "    files = {'data': buffer}\n",
    "    \n",
    "    print(f\"Uploading data field '{data_field_id}'...\")\n",
    "    try:\n",
    "        response = session.post(upload_url, data=data, files=files)\n",
    "        if response.status_code == 201:\n",
    "            print(\"Data field created successfully.\")\n",
    "        else:\n",
    "            print(f\"Error creating data field: {response.status_code}\")\n",
    "            print(response.text)\n",
    "        response.raise_for_status()\n",
    "    except Exception as e:\n",
    "        print(f\"Upload failed: {e}\")\n",
    "\n",
    "# ---------------------------------------------------------\n",
